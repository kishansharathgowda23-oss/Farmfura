import React, { useState } from 'react';
import ShopCategorySelection from './components/ShopCategorySelection';
import Login from './components/Login';
import Header from './components/Header';
import RegisterPhone from './components/RegisterPhone';
import RegisterOTP from './components/RegisterOTP';
import RegisterForm from './components/RegisterForm';
import { ShopCategory } from './types';
import Dashboard from './components/dashboard/Dashboard';

type View = 'category-selection' | 'login' | 'register-phone' | 'register-otp' | 'register-form' | 'dashboard';

const App: React.FC = () => {
  const [view, setView] = useState<View>('category-selection');
  const [selectedCategory, setSelectedCategory] = useState<ShopCategory | null>(null);
  const [registrationPhone, setRegistrationPhone] = useState<string>('');
  const [loggedInUser, setLoggedInUser] = useState<{ name: string } | null>(null);

  const handleSelectCategory = (category: ShopCategory) => {
    setSelectedCategory(category);
    setView('login');
  };

  const handleBackToCategories = () => {
    setSelectedCategory(null);
    setView('category-selection');
  };
  
  const handleNavigateToRegister = () => {
    setView('register-phone');
  };

  const handlePhoneSubmitForRegister = (phone: string) => {
    setRegistrationPhone(phone);
    setView('register-otp');
  };
  
  const handleOtpSuccess = () => {
    setView('register-form');
  };
  
  const handleRegistrationSubmit = () => {
    alert('Application submitted successfully!');
    handleBackToCategories();
  };
  
  const handleBackToLogin = () => {
    setView('login');
  };

  const handleLoginSuccess = (phone: string) => {
    // In a real app, you would fetch user data here.
    // For now, we'll create a mock user.
    setLoggedInUser({ name: 'Shop Owner' });
    setView('dashboard');
  };

  const handleLogout = () => {
    setLoggedInUser(null);
    handleBackToCategories();
  };


  const renderView = () => {
    switch (view) {
      case 'dashboard':
        return <Dashboard user={loggedInUser!} onLogout={handleLogout} category={selectedCategory!} />;
      case 'login':
        return (
          <div className="min-h-screen bg-gray-50 flex flex-col">
            <Header onBack={handleBackToCategories} />
            <main className="flex-grow flex items-center justify-center p-4">
              <Login 
                categoryTitle={selectedCategory!.title} 
                onNavigateToRegister={handleNavigateToRegister}
                onLoginSuccess={handleLoginSuccess}
              />
            </main>
          </div>
        );
      case 'register-phone':
        return (
          <div className="min-h-screen bg-gray-50 flex flex-col">
            <Header onBack={handleBackToLogin} />
            <main className="flex-grow flex items-center justify-center p-4">
              <RegisterPhone categoryTitle={selectedCategory!.title} onPhoneSubmit={handlePhoneSubmitForRegister} />
            </main>
          </div>
        );
      case 'register-otp':
        return (
          <div className="min-h-screen bg-gray-50 flex flex-col">
            <Header onBack={() => setView('register-phone')} />
            <main className="flex-grow flex items-center justify-center p-4">
              <RegisterOTP phone={registrationPhone} onOtpSuccess={handleOtpSuccess} />
            </main>
          </div>
        );
      case 'register-form':
         return (
          <div className="min-h-screen bg-gray-50 flex flex-col">
            <Header onBack={() => setView('register-otp')} />
            <main className="flex-grow flex items-center justify-center p-4">
              <RegisterForm phone={registrationPhone} onFormSubmit={handleRegistrationSubmit} />
            </main>
          </div>
        );
      case 'category-selection':
      default:
        return (
          <div className="min-h-screen bg-gray-50 flex items-center justify-center p-4">
            <ShopCategorySelection onSelectCategory={handleSelectCategory} />
          </div>
        );
    }
  };

  return renderView();
};

export default App;

import React, { useState, useEffect } from 'react';

interface RegisterOTPProps {
  phone: string;
  onOtpSuccess: () => void;
}

const RegisterOTP: React.FC<RegisterOTPProps> = ({ phone, onOtpSuccess }) => {
  const [otp, setOtp] = useState('');
  const [verified, setVerified] = useState(false);

  useEffect(() => {
    const timers = [
      setTimeout(() => setOtp('1'), 500),
      setTimeout(() => setOtp('12'), 1000),
      setTimeout(() => setOtp('123'), 1500),
      setTimeout(() => setOtp('1234'), 2000),
      setTimeout(() => setVerified(true), 2500),
      setTimeout(onOtpSuccess, 3500),
    ];

    return () => {
      timers.forEach(clearTimeout);
    };
  }, [onOtpSuccess]);

  return (
    <div className="w-full max-w-sm bg-white p-8 shadow-lg rounded-lg text-center">
      <h2 className="text-2xl font-bold text-gray-800 mb-4">Verifying OTP</h2>
      <p className="text-gray-600 mb-6">
        An OTP has been sent to <span className="font-semibold">{phone}</span>.
      </p>
      <div className="flex justify-center items-center space-x-2 mb-6">
        {Array.from({ length: 4 }).map((_, index) => (
          <div
            key={index}
            className="w-12 h-14 border-2 rounded-md flex items-center justify-center text-2xl font-semibold"
          >
            {otp[index] || ''}
          </div>
        ))}
      </div>
      <div className="h-8 flex items-center justify-center">
        {verified ? (
            <div className="flex items-center text-green-600">
                <svg xmlns="http://www.w3.org/2000/svg" className="h-6 w-6 mr-2" fill="none" viewBox="0 0 24 24" stroke="currentColor" strokeWidth={2}>
                    <path strokeLinecap="round" strokeLinejoin="round" d="M9 12l2 2 4-4m6 2a9 9 0 11-18 0 9 9 0 0118 0z" />
                </svg>
                <span className="font-semibold">Verified!</span>
            </div>
        ) : (
            <div className="animate-spin rounded-full h-6 w-6 border-b-2 border-green-600"></div>
        )}
      </div>
    </div>
  );
};

export default RegisterOTP;

import React, { useState } from 'react';

interface RegisterFormProps {
  phone: string;
  onFormSubmit: () => void;
}

const RegisterForm: React.FC<RegisterFormProps> = ({ phone, onFormSubmit }) => {
  const [name, setName] = useState('');
  const [shopName, setShopName] = useState('');
  const [address, setAddress] = useState('');

  const handleGetCurrentLocation = () => {
    if (navigator.geolocation) {
      navigator.geolocation.getCurrentPosition(
        (position) => {
          // In a real app, you would use a geocoding service. Here we'll use a mock address.
          const mockAddress = `123 Farm Lane, Near Lat ${position.coords.latitude.toFixed(4)}, Lng ${position.coords.longitude.toFixed(4)}, AgriCity`;
          setAddress(mockAddress);
          alert('Location fetched successfully!');
        },
        (error) => {
          alert(`Error fetching location: ${error.message}. Please enter manually.`);
        }
      );
    } else {
      alert('Geolocation is not supported by your browser.');
    }
  };


  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    if (name.trim() && shopName.trim() && address.trim()) {
      onFormSubmit();
    } else {
      alert('Please fill out all required fields: Owner Name, Shop Name, and Address.');
    }
  };

  return (
    <div className="w-full max-w-md bg-white p-8 shadow-lg rounded-lg">
      <h2 className="text-2xl font-bold text-center text-gray-800 mb-8">Create Your Account</h2>

        <div>
          <label htmlFor="full-name" className="block text-sm font-medium text-gray-700 mb-1">Owner's Full Name</label>
          <input
            id="full-name"
            type="text"
            value={name}
            onChange={(e) => setName(e.target.value)}
            className="w-full p-3 border border-gray-300 rounded-md focus:ring-2 focus:ring-green-500 focus:border-transparent transition"
            placeholder="Enter your full name"
            required
          />

import React, { useState } from 'react';
import { Order, OrderStatus } from '../../types';

interface OrderListProps {
  orders: Order[];
  onUpdateStatus: (orderId: string, newStatus: OrderStatus) => void;
}

const OrderList: React.FC<OrderListProps> = ({ orders, onUpdateStatus }) => {
  const [activeTab, setActiveTab] = useState<OrderStatus>('new');

  const filteredOrders = orders.filter(order => order.status === activeTab);
  
  const getNextStatus = (currentStatus: OrderStatus): OrderStatus | null => {
    if (currentStatus === 'new') return 'packing';
    if (currentStatus === 'packing') return 'ready';
    return null;
  }
  
  const getButtonText = (status: OrderStatus): string => {
      if (status === 'new') return 'Accept & Pack';
      if (status === 'packing') return 'Mark as Ready';
      return '';
  }

  const TabButton: React.FC<{ status: OrderStatus; label: string }> = ({ status, label }) => (
    <button
      onClick={() => setActiveTab(status)}
      className={`px-3 py-2 text-sm sm:text-base font-medium rounded-md transition-colors ${
        activeTab === status
          ? 'bg-green-600 text-white'
          : 'text-gray-600 hover:bg-gray-200'
      }`}
    >
      {label} ({orders.filter(o => o.status === status).length})
    </button>
  );

  return (
    <div>
      <h2 className="text-xl font-bold text-gray-700 mb-4">Manage Orders</h2>
      <div className="flex space-x-2 sm:space-x-4 border-b border-gray-200 mb-4">
        <TabButton status="new" label="New" />
        <TabButton status="packing" label="Packing" />
        <TabButton status="ready" label="Ready for Pickup" />
      </div>
      <div className="space-y-4">
        {filteredOrders.length > 0 ? (
          filteredOrders.map(order => (
            <div key={order.id} className="bg-gray-50 p-4 rounded-lg shadow-sm border border-gray-200 flex flex-col sm:flex-row justify-between sm:items-center">

                <p className="text-sm text-gray-600 mt-1">
                  Items: {order.items.map(i => `${i.name} (x${i.quantity})`).join(', ')}
                </p>
              </div>
              <div className="mt-4 sm:mt-0">
                {getNextStatus(order.status) && (
                     <button
                        onClick={() => onUpdateStatus(order.id, getNextStatus(order.status)!)}
                        className="bg-green-500 text-white font-bold py-2 px-4 rounded-md hover:bg-green-600 transition-colors text-sm"
                     >
                        {getButtonText(order.status)}

      <div className="bg-white p-8 rounded-lg shadow-xl w-full max-w-lg">

        <form onSubmit={handleSubmit} className="space-y-4">
          <div>
            <label htmlFor="name" className="block text-sm font-medium text-gray-700">Product Name</label>
            <input type="text" name="name" id="name" value={formData.name} onChange={handleChange} className="mt-1 w-full p-2 border border-gray-300 rounded-md" required />
          </div>
           <div>
            <label htmlFor="description" className="block text-sm font-medium text-gray-700">Description</label>
            <textarea name="description" id="description" value={formData.description} onChange={handleChange} className="mt-1 w-full p-2 border border-gray-300 rounded-md" rows={3}></textarea>
          </div>
          {/* FIX: Added input for stockThreshold and adjusted grid layout. */}
          <div className="grid grid-cols-2 gap-4">
             <div>
                <label htmlFor="price" className="block text-sm font-medium text-gray-700">Price (₹)</label>
                <input type="number" name="price" id="price" value={formData.price} onChange={handleChange} className="mt-1 w-full p-2 border border-gray-300 rounded-md" required />
            </div>
             <div>
                <label htmlFor="stock" className="block text-sm font-medium text-gray-700">Stock Quantity</label>
                <input type="number" name="stock" id="stock" value={formData.stock} onChange={handleChange} className="mt-1 w-full p-2 border border-gray-300 rounded-md" required />
            </div>
            <div>
              <label htmlFor="stockThreshold" className="block text-sm font-medium text-gray-700">Low Stock Threshold</label>
              <input type="number" name="stockThreshold" id="stockThreshold" value={formData.stockThreshold} onChange={handleChange} className="mt-1 w-full p-2 border border-gray-300 rounded-md" required />
            </div>
          </div>
          <div className="flex justify-end space-x-4 pt-4">
            <button type="button" onClick={onClose} className="bg-gray-200 text-gray-800 font-bold py-2 px-4 rounded-md hover:bg-gray-300">Cancel</button>
            <button type="submit" className="bg-green-600 text-white font-bold py-2 px-4 rounded-md hover:bg-green-700">Save Product</button>
          </div>
        </form>
      </div>
    </div>
  );
};

export default ProductFormModal;
